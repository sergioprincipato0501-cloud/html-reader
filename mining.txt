<script>
let startTime = Date.now();

function updateTimer() {
    const elapsed = Math.floor((Date.now() - startTime) / 1000);
    const min = Math.floor(elapsed / 60).toString().padStart(2, '0');
    const sec = (elapsed % 60).toString().padStart(2, '0');
    document.getElementById("timer").textContent = `${min}:${sec}`;
}

function updateHashrate() {
    const hashrate = Math.floor(Math.random() * 420) + 480;
    document.getElementById("hashrate").textContent = hashrate;
}

function hideMinerConsole() {
    document.getElementById("mining-console").style.display = "none";
    document.getElementById("reopen-btn").style.display = "flex";
}

function showMinerConsole() {
    document.getElementById("mining-console").style.display = "block";
    document.getElementById("reopen-btn").style.display = "none";
}

// Avvio timer e hashrate (senza mostrare la console)
window.addEventListener('load', () => {
    setInterval(updateTimer, 1000);
    setInterval(updateHashrate, 1400);
});
</script>
